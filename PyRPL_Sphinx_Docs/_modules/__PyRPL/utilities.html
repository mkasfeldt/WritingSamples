import os
from time import time,localtime
from functools import wraps
from decimal import Decimal

def timed(f):
  @wraps(f)
  def wrapper(*args, **kwds):
    """decorator to show the time to complete a function"""
    #print 'start timer'
    start = time()
    result = f(*args, **kwds)
    elapsed = time() - start
    print "%s took %d seconds to finish" % (f.__name__, elapsed)
    return result
  return wrapper


[docs]def fixUnicode(s):
    """
    fix unicode issues with data from mssql ACS database.  it is in cp1252
    format.  sqlite can only be in utf-8 or utf-16.  this code checks to
    see if the current unicode will convert cleanly to utf-8.  if it does
    not it forces the non-acceptable utf codes to use accpetable
    unicode characters.

    :param s: unicode data to be cleaned
    :type s: unicode
    :return: valid utf-8 data

    

    """
    try:
        temp = s.decode('utf-8') #if not fix it to be acceptable utf-8 code.
    except:
        temp =''
        halftemp = unicode(s,'cp1252')
        if halftemp.find(u'\xbd')>-1:                
            s = str(halftemp.replace(u'\xbd','half'))
        if halftemp.find(u'\u2013')>-1:
            s = str(halftemp.replace(u'\u2013','-'))
        temp = unicode(s,'utf-8','replace')
        s = temp.replace(u'\ufffd',"'")
    return s

[docs]def fixStreetAddress(address):
  """address must be a list with the data in the following order.

  1. N_NUMBER_PREFIX.
  2. N_NUMBER.
  3. N_NUMBER_SUFFIX.
  4. N_PRE_DIRECTIONAL.
  5. N_STREET_NAME.
  6. N_POST_DIRECTIONAL.
  7. N_STREET_TYPE.
  8. N_UNIT_TYPE.
  9. N_UNIT_NUMBER.
  
  This is used to build addresses for data from ACS tables. this function
  returns a tuple containing the formatted full street address, street number,
  street and unit.

  """
  #clear nulls
  x = 0
  for i in address:
    if i == None:
      address[x] = u""
    else:
      address[x] = fixUnicode(i)
    x = x +  1
    
  #build Address
  addr ='%s%s%s'%(address[0],address[1],address[2])
  snumber = addr.strip(" ") #only address number for the fire number field
  
  s = '%s %s'%(address[3],address[4],)
  s = s.strip(" ")
  street = s
  addr = addr +" "+ s
  addr = addr.strip(" ")
  
  s = '%s %s'%(address[5],address[6])
  s = s.strip(" ")
  street = street + " "+ s #only the street name
  street = street.strip(" ")
  addr = addr +" "+ s      
  addr = addr.strip(" ")
  
  s = " %s %s"%(address[7],address[8])
  s = s.strip(" ")
  unit = s #only the unit type
  addr = addr +" "+ s
  addr = addr.strip(" ") #full street adress line
  return (addr,snumber,street,unit)
