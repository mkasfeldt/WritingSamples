[docs]def updateDB(path=None,startyear=2010,futureyear=False,update=False):
    """
    Used to update the RPL and tax databases. If path is blank the location
    of the script is the location of the database. startyear will begin
    with 2010 unleses otherwise specified by using a 4-digit year as an integer.
    if futureyear is set to True it will create records for the next rollyear
    (i.e. if it is 2012 it will create 2013). update denotes if a temporary
    database should be used.
    
    """
    
    from __PyRPL import acs_taxes, acs_property, acs_persprop
    #property Database creation
    if update:
        z = acs_property.Property(dbname='workingproperty.sqlite')
    else:
        z = acs_property.Property()
    
    #set range of years starting at startyear
    lastyear = datetime.now().year + 1
    if futureyear:
        lastyear = lastyear + 2
    
    #update the property database
    for year in range(startyear,lastyear):
        z.createRollYear(year)

    #set Taxdata
    if update:
        z = acs_taxes.Taxdata(dbname='workingtaxbill.sqlite')
    else:
        z = acs_taxes.Taxdata()
    
    #set last tax year to 1 less than rollyear
    fmod = 0
    if futureyear:
        fmod = 1        
    lastyear = lastyear - 1 - fmod
    z.updateTaxDB(lastyear)
