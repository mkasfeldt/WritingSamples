[docs]def makeGISfile(rollyear,taxyear=None,dbpath=None,filepath=None):
  """
  create a file for the gis system and return data. rollyear is the four digit
  year of the roll year you wish to export into the tab.gis file, if it is not
  set it will default to the current year.  taxyear is the four digit year
  that will be used to get the tax data from the property database
  (current year taxes, previous year taxes, lottery credit count and lottery
  credit), if not set it will be the same year as the roll year.  dbpath is
  the path to the property.sqlite database, if not set it will be the location
  of the script running this function.  filepath is the path where the gis.tab
  file will be created dbpath is same location as script"""
  #set the path to the database and ensure it is linux friendly
  if dbpath:
    lconn = sqlite3.connect('%s/property.sqlite'%dbpath.replace('\\','/'))
  else:
    lconn = sqlite3.connect('%s/property.sqlite'%os.getcwd())
  if filepath:
    filepath = filepath.replace('\\','/')
  else:
    filepath ='.'
  #set the tax year if it is not set
  if not taxyear:
    taxyear = rollyear
  lc = lconn.cursor()
  #write the header to the gis.tab file
  gis =[]
  header="PIN\tCOMPUTER_NO\tOWNER_NAME\tFIRST_NAME\tLAST_NAME\tSTREET\tCITY\tSTATE\tZIP\tPAR_ADDRESS\tPAR_ZIP_CODE\tPAR_OCCUPANCY\tPAR_FIRE_NO\tPAR_STREET_INDEX\tZONING1\tZONING2\tZONING3\tMUN_NO\tSECTION\tTOWN\tRANGE\tLEGAL_DESC\tLEGAL_ACRES\tSCHOOL_DIST\tSCHOOL_CODE\tDISTRICT1\tDISTRICT2\tCOMPUTER_ALPHA\tPIN_KEY\tHISTORY\tPAR_NUMBER\tDOCUMENT_NO\tOLD_COMP_NO\tCURRENT_TAXES_BEFORE_LOTTERY_CREDIT\tPREVIOUS_TAXES_BEFORE_LOTTERY_CREDIT\tLOTTERY\tLOTTERY_AMT\tCOUNT\tTOT_LAND\tTOT_IMPROVE\tTOT_VALUE\n"
  f = open('%s/gis.tab'%filepath,'w')
  f.write(header)
  #get base data
  print 'select data from new property'
  ssql = ("select PIN,AltNumber,DeededName,Legal,Acres,UTA,History,MapID,DocumentNumber,CurrentYearTaxes,"
          "PreviousYearTaxes,LotteryCreditCount,LotteryCredit,AssessedLand, "
          "AssessedImprovement,AssessedTotal "
          "from newProperty "
          "where rollyear = %s"%rollyear)

  
  try:
    p = lc.execute(ssql).fetchall()
  except:
    print ssql
    raise 
  print 'ready to start'
  print len(p)
  prop = {}
  for i in p:
    if prop.has_key(i[0]):
      print i[0]
    else:
      prop[i[0]] = i
  #if the rollyear and tax year are not equal update the tax data fields in the prop dict   
  if taxyear != rollyear:
    ssql = ("select PIN,CurrentYearTaxes,"
            "PreviousYearTaxes,LotteryCreditCount,LotteryCredit "
            "from newProperty "
            "where rollyear = %s"%taxyear)
    #make sure the sql statement worked if it does not fail gracefully and continue
    try:
      p2 = lc.execute(ssql).fetchall()
    except:
      print ssql
      raise
    #update the prop dict here
    for i in p2:
      if prop.has_key(i[0]):
        prop[i[0]][9] = i[1]
        prop[i[0]][10] = i[2]
        prop[i[0]][11] = i[3]
        prop[i[0]][12] = i[4]
      else:
        prop[i[0]][9] = i[0]
        prop[i[0]][10] = i[0]
        prop[i[0]][11] = i[0]
        prop[i[0]][12] = i[0]
  print len(prop)
  print'prop base done'
  #get mail to name and address
  ssql =("select PIN,FirstName,LastName,address,City,State,zipcode,percent from newName where rollyear =%s and mailto='T'"%rollyear)
  n  = lc.execute(ssql).fetchall()
  name = {}
  print len(n)
  for i in n:
    if name.has_key(i[0]):
      if name[i[0]][7] < i[7]:
        name[i[0]] = i
    else:
      name[i[0]] = i
  print len(name)
  print 'name base done'
  #get physical address
  ssql =("select PIN,address,zipcode,streetnumber from newAddress where rollyear= %s"%rollyear)
  a = lc.execute(ssql).fetchall()
  print len(a)
  address = {}
  for i in a:
    if address.has_key(i[0]):
      print i[0]
    else:
      address[i[0]] = i
  print len(address)
  print 'address base done'
  #get zoning info
  ssql = "select pin, zone from newZoning where rollyear=%s"%rollyear
  z = lc.execute(ssql).fetchall()
  zone = {}
  print len(z)

  x = 0 
  for i in z:        
    if zone.has_key(i[0]):
      tlist = zone[i[0]]
      tlist.append(i[1])
      zone[i[0]] = tlist           
    else:
      zone[i[0]] = [i[1]]      

    x = x + 1
  print len(zone)
  print ('zone base done')
  
  #start making the list for the gis file. the data is from the header. when a 0 is inserted it mean the data is not captured by the ACS system and is not used but needs
  #to be included in the file so it can be imported into the current gis system
  for k,v in prop.iteritems():
    #get name
    if name.has_key(k):
      nr = name[k]
    else:
      nr = 0
    #get address
    if address.has_key(k):
      ar = address[k]
    else:
      ar =0
    #get zone
    if zone.has_key(k):
      zr = zone[k]
    else:
      zr = 0
    #create the giant 'fing string
    #pin alt and mail to
    if nr:
      s ='"%s"\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t%s\t'%(v[0],v[1],v[2],nr[1],nr[2],nr[3],nr[4],nr[5],nr[6])
    else:
      s ='%s\t%s\t%s\t\t\t\t\t\t\t'%(v[0],v[1],v[2])
    
    #parcel address and fillers
    if ar:
      s ='%s%s\t%s\t0\t%s\t0\t'%(s,ar[1],ar[2],ar[3])
    else:
      s ='%s\t\t0\t\t0\t'%s
    
    #add zoning info the file up aw_input('kill')to to 3 zone
    if zr:
      x = 0
      for i in zr:
        if x <3:
          s ='%s%s\t'%(s,i)
        x = x + 1
      #add nulls if needed
      addnozone = 3 - x
      if x >0:
        for j in range(0,addnozone):
          s = '%s\t'%s
    else:
      s = '%s\t\t\t'%s
    #section town range
    s = '%s%s\t%s\t%s\t%s\t'%(s,v[0][2:5],v[0][10:12],v[0][6:8],v[0][8:10])
    #legal acres thru district 2
    s = '%s%s\t%s\t%s\t%s\t0\t0\t'%(s,v[3],v[4],v[5],v[5])
    #comp alpah thru oldcompno
    s = '%s%s\t%s\t%s\t%s\t%s\t%s\t'%(s,v[1],v[0][6:],v[6],v[7],v[8],v[1])
    #tax to end
    s = '%s%s\t%s\t%s\t%s\t0\t%s\t%s\t%s'%(s,v[9],v[10],v[11],v[12],v[13],v[14],v[15])
    s = s.replace('\n',' ')
    s = s.replace('\r',' ')
    s = '%s\n'%s
    gis.append(s)
    f.write(s)
  f.close()
  return gis
